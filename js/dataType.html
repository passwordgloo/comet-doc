    # 字符串含义

    字符串使用英文状态下的" "或' ' 单引号和双引号可以相互嵌套，但注意不能自己嵌套自己

    # 字符串拼接

let comic = '火影忍者'
let author = '岸本齐史'
document.write(comic + author)
    # 模板字符串

document.write(${comic}的作者是${author})
