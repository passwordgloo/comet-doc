                      # 求数组所有元素的和以及平均值

                      let arr = [2,1,35,43]
let sum = 0
let average  = 0
for (let i = 0 ;i <arr.length;i++){
    //i输出的是数组下标
    //arr[i]是数组元素
    sum += arr[i]
}
average = sum / arr.length
document.write(`总分：${sum},平均分${average}`)
